import { generateText } from 'ai'
import { anthropic } from '@ai-sdk/anthropic'
import { v0Tools } from '@v0-sdk/ai-tools'
import 'dotenv/config' 

async function startAntigravityAgent() {
  console.log("🌀 Antigravity Environment Initialized...");
  console.log("🤖 Booting Claude Fable 5 Agent Loop...");

  const result = await generateText({
    // 1. Direct Fable 5 to handle the deep reasoning loop
    model: anthropic('claude-fable-5'), 
    
    // 2. Define the core task layout for the agent
    prompt: 'Build and deploy a beautiful dark-mode React dashboard for a personal project using the v0 tools.',
    
    // 3. Mount the v0 deployment tool suite fueled by your credits
    tools: v0Tools({ apiKey: process.env.V0_API_KEY }),
    
    // 4. Give the agent up to 10 step loops to think, build, and resolve errors
    maxSteps: 10, 
  })

  console.log("✨ Goal Achieved! Agent Summary:")
  console.log(result.text)
}

startAntigravityAgent()
