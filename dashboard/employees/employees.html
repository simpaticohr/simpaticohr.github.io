window.saveEmployee = async function(e) { if (e) e.preventDefault(); var form = document.getElementById('addEmployeeForm'); if (!form.checkValidity()) { form.reportValidity(); return; } // 1. Build the payload matching your Supabase snake_case columns var newEmp = { first_name: document.getElementById('empFirstName').value.trim(), last_name: document.getElementById('empLastName').value.trim(), email: document.getElementById('empEmail').value.trim(), department: document.getElementById('empDept').value, // Adjust if you use department_id job_title: document.getElementById('empTitle').value.trim(), start_date: document.getElementById('empStartDate').value || null, type: document.getElementById('empType').value, location: document.getElementById('empLocation').value.trim(), status: 'active' }; // 2. Connect to your existing Supabase client const client = window.sb ? window.sb() : (window.supabase ? window.supabase.createClient(OPS_CONFIG.supabaseUrl, OPS_CONFIG.supabaseKey) : null); if (!client) { if (window.showToast) window.showToast('Database connection missing', 'error'); return; } try { // 3. Send the actual INSERT request to Supabase const { data, error } = await client.from('employees').insert([newEmp]).select(); if (error) throw error; if (window.showToast) window.showToast('✅ Employee saved to database!'); // 4. Close the modal safely var modalEl = document.getElementById('addEmployeeModal'); var modal = bootstrap.Modal.getInstance(modalEl); if (modal) { modal.hide(); } else { modalEl.classList.remove('show'); modalEl.style.display = 'none'; document.body.classList.remove('modal-open'); var backdrop = document.querySelector('.modal-backdrop'); if (backdrop) backdrop.remove(); } form.reset(); // 5. Re-fetch from Supabase to update the table if (window.loadEmployees) { await window.loadEmployees(); } else { window.location.reload(); // Fallback if dynamic reload isn't hooked up } } catch (err) { console.error('Supabase Insert Error:', err); if (window.showToast) window.showToast(`❌ Error: ${err.message}`, 'error'); } };
